# Examples

## Basic

This is a card component for demo.

```typescript
type Props = {
  data: any[];
  index: number;
  style: React.CSSProperties;
};

const SamplCard: React.FC<Props> = ({ data, index, style }) => (
  <div style={{ ...style, backgroundColor: '#fff' }}>
    <div style={{ padding: 16 }}>
      <h2>{data[index]}</h2>
    </div>
  </div>
);
```

CardWindow provide children with props.

```typescript
type Props = {
  data: number[];
  cardRect: { width: number; height: number };
  styles: { root?: React.CSSProperties; scroll?: React.CSSProperties };
};

const BasicExample: React.FC<Props> = ({ data, cardRect, styles }) => (
  <CardWindow data={data} cardRect={cardRect} styles={styles}>
    {SampleCard}
  </CardWindow>
);
```

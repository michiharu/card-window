Styleâ€‹

By default, CardWindow has the style { width: '100%', height: '100%' }. These styles can be overridden.

Live Editor
function App() {
  const items = [...Array(1000)].map((_, i) => i);
  const cardRect = { width: 300, height: 200 };
  const style = { height: 400 };
  return (
    <CardWindow data={items} cardRect={cardRect} style={style}>
      {({ index, style }) => (
        <div className="card" style={{ ...style, height: style.height - 16 }}>
          <div style={{ padding: 16 }}>
            <h2>{index}</h2>
          </div>
        </div>
      )}
    </CardWindow>
  );
}
